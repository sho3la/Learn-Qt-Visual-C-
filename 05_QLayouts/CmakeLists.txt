cmake_minimum_required(VERSION 3.9)

project ("05_QLayouts")

# list the header files
set(HEADER_FILES

)

# list the source files
set(SOURCE_FILES
	main.cpp
)


# link with qt5
find_package(Qt5Core    CONFIG REQUIRED)
find_package(Qt5Widgets CONFIG REQUIRED)
find_package(Qt5Gui     CONFIG REQUIRED)

qt5_wrap_cpp(HEADER_MOC ${HEADER_FILES})

# add library target
add_executable(05_QLayouts
	${HEADER_FILES}
	${SOURCE_FILES}
	${HEADER_MOC}
	${RCC_SOURCES}
)

set(QT_LIBS
	Qt5::Core
	Qt5::Gui
	Qt5::Widgets
)

target_link_libraries(05_QLayouts
	PUBLIC
		${QT_LIBS}
)


# Windows specfic flags to enable W variations of winapi
if(WIN32)
	target_compile_definitions(05_QLayouts
		PRIVATE
			OS_WINDOWS=1
	)
	target_compile_options(05_QLayouts
		PRIVATE
			/WX
	)
endif()

# enable C++17
# disable any compiler specifc extensions
target_compile_features(05_QLayouts PUBLIC cxx_std_17)
set_target_properties(05_QLayouts PROPERTIES
	CXX_EXTENSIONS OFF
)
